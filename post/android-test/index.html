public class WinningPrize {

    /**
     * 单次中奖概率
     */
    private double winningProbability;

    /**
     * 一轮游戏连续玩几次
     */
    private int playTimes;

    /**
     * 多少轮
     */
    private int N;

    private int winCount;


    public WinningPrize(double winningProbability, int playTimes, int n) {
        this.winningProbability = winningProbability;
        this.playTimes = playTimes;
        N = n;
    }


    public void run() {
        for (int i = 0; i < N; i++) {
            // N轮游戏
            if (play()) {
                winCount++;
            }
        }

        System.out.println("winning rate : " + (double) winCount / N);
    }

    /**
     * 一轮游戏
     * playTimes次
     */
    private boolean play() {
        for (int i = 0; i < playTimes; i++) {
            // 随机数，如果比winningProbability小，认为赢了
            if (Math.random() < winningProbability) {
                return true;
            }
        }

        return false;
    }


    public static void main(String[] args) {

        double winningProbability = 0.2;
        int playTimes = 5;
        int N = (int) 1e8;

        WinningPrize winningPrize = new WinningPrize(winningProbability, playTimes, N);
        winningPrize.run();
    }
}
